```javascript
// UserOperationの作成例
import { NeroUserOpSDK } from '@nerochain/user-op-sdk';

// SDKの初期化
const userOpSDK = new NeroUserOpSDK({
  rpcUrl: 'https://testnet-rpc.nerochain.io',
  entryPointAddress: '0x...',
});

// UserOperationの作成
async function createUserOperation() {
  const userOp = await userOpSDK.createUserOp({
    target: '0xTargetAddress',
    data: '0xCallData',
    value: '0',
    // その他のパラメータ...
  });
  
  return userOp;
}
```
