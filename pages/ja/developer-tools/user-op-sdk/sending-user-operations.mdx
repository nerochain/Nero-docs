```javascript
// UserOperationの送信例
import { NeroUserOpSDK } from '@nerochain/user-op-sdk';

// SDKの初期化
const userOpSDK = new NeroUserOpSDK({
  rpcUrl: 'https://testnet-rpc.nerochain.io',
  entryPointAddress: '0x...',
});

// UserOperationの送信
async function sendUserOperation(userOp) {
  const txHash = await userOpSDK.sendUserOp(userOp);
  console.log('Transaction Hash:', txHash);
  return txHash;
}
```
