```javascript
// ウォレット連携の例
import { NeroUserOpSDK, WalletAdapter } from '@nerochain/user-op-sdk';

// ウォレットアダプターの設定
const walletAdapter = new WalletAdapter({
  type: 'metamask',
  // その他のウォレット固有の設定...
});

// SDKの初期化（ウォレットアダプター付き）
const userOpSDK = new NeroUserOpSDK({
  rpcUrl: 'https://testnet-rpc.nerochain.io',
  entryPointAddress: '0x...',
  walletAdapter
});
```
