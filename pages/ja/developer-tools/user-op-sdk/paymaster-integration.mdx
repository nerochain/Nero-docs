```javascript
// ペイマスター統合の例
import { NeroUserOpSDK, PaymasterConfig } from '@nerochain/user-op-sdk';

// ペイマスター設定
const paymasterConfig: PaymasterConfig = {
  url: 'https://paymaster-api.nerochain.io',
  apiKey: 'YOUR_API_KEY',
  context: {
    // ペイマスターに必要なコンテキスト情報
    sponsorshipPolicy: 'gas-only',
    // その他の情報...
  }
};

// SDKの初期化（ペイマスター付き）
const userOpSDK = new NeroUserOpSDK({
  rpcUrl: 'https://testnet-rpc.nerochain.io',
  entryPointAddress: '0x...',
  paymasterConfig
});
