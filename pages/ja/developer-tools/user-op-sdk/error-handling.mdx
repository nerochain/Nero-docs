```javascript
// エラー処理の例
import { NeroUserOpSDK, UserOpError } from '@nerochain/user-op-sdk';

// SDKの初期化
const userOpSDK = new NeroUserOpSDK({
  rpcUrl: 'https://testnet-rpc.nerochain.io',
  entryPointAddress: '0x...',
});

// エラー処理を含むUserOperationの送信
async function sendUserOpWithErrorHandling(userOp) {
  try {
    const txHash = await userOpSDK.sendUserOp(userOp);
    console.log('Transaction Hash:', txHash);
    return txHash;
  } catch (error) {
    if (error instanceof UserOpError) {
      // UserOp固有のエラー処理
      console.error('UserOp Error:', error.code, error.message);
    } else {
      // その他のエラー処理
      console.error('Unexpected Error:', error);
    }
    throw error;
  }
}
